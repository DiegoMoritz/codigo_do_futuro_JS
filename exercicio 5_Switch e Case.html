let nome = prompt("Digite o nome") let valor = Number(prompt(` Olá ${nome} Digite o valor que você tem para a viagem `)) let opcaoViagem = parseInt(prompt(` Escolha um dos destinos abaixo: 1) Disney 2) Madrid 3) Maceió 4) Ribeirão preto `)) switch(opcaoViagem) { case 1: valorViagem = 15000 break; case 2: valorViagem = 12000 break; case 3: valorViagem = 8000 break; case 4: valorViagem = 5000 break; } var parar = 0 if (opcaoViagem < 1 || opcaoViagem > 4) { alert("Opção inválida") parar = 1 } let valorSobraCliente = valor - valorViagem if (valorSobraCliente >= 0 && parar != 1){ alert(`Parabéns ${nome}, você irá conseguir fazer a viagem e o seu troco é de: ${ valorSobraCliente.toLocaleString('pt-br',{style: 'currency', currency: 'BRL'}) }`) } else if (valorSobraCliente < 0 && parar != 1){ valorSobraCliente = valorSobraCliente * -1 alert(`Infelizmente você não tem saldo suficiente para fazer a viagem: falta ${ valorSobraCliente.toLocaleString('pt-br',{style: 'currency', currency: 'BRL'}) }`) }
